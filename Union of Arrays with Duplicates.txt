You are given two arrays a[] and b[], 
return the Union of both the arrays in any order.
code:
    vector<int> findUnion(vector<int>& a, vector<int>& b) {
        set<int> st1;
        vector<int> temp;
        for(int i=0;i<a.size();i++){
            st1.insert(a[i]);
        }
        for(int i=0;i<b.size();i++){
            st1.insert(b[i]);
        }
        for(auto it:st1){
            temp.push_back(it);
        }
        return temp;
    }
